# 显示行号
:set number
:set nu!

# 排序
:[range]sor[t][!] [b][f][i][n][o][r][u][x] [/{pattern}/]
:help sort
:sort
:sort!      逆序
:sort i     忽略大小写

# 跳转到光标的上/下次位置
Ctrl+O
Ctrl+I

# 替换
:%s/pattern/string/g
# 删除空白行
:g/^\s*$/d
# 删除偶数行
:g/^/+1 d
:%norm jdd
# 删除奇数行
:g/^/d|m
:%norm jkdd

# 将ascii模式转换成16进制字符
:%!xxd
# 还原原先模式
:%!xxd -r
